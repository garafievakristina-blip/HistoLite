// ================================================================ // АНАЛИТИКА: СВЯЗЬ СИМПТОМОВ С ФАЗОЙ ЦИКЛА // ================================================================ function renderCycleSymptomAnalysis() { const container = document.getElementById('cycle-symptom-analysis'); if (!container) return; if (!userLogs || userLogs.length === 0) { container.innerHTML = '

Нет записей для анализа.

'; return; } // Группируем записи по фазам цикла const phases = { 'Менструация': { days: [], symptoms: {} }, 'Фолликулярная': { days: [], symptoms: {} }, 'Овуляция': { days: [], symptoms: {} }, 'Лютеиновая': { days: [], symptoms: {} }, 'Стабильная': { days: [], symptoms: {} } }; // Определяем фазу по дню цикла function getPhase(day) { if (!day) return 'Стабильная'; if (day <= savedPeriodDays) return 'Менструация'; const ovulationDay = savedCycleLength - 14; if (day === ovulationDay) return 'Овуляция'; if (day >= ovulationDay - 5 && day <= ovulationDay + 1) return 'Овуляция'; if (day > ovulationDay + 1 && day < savedCycleLength) return 'Лютеиновая'; if (day >= savedCycleLength) return 'Лютеиновая'; // задержка считаем лютеиновой return 'Фолликулярная'; } userLogs.forEach(log => { if (!log.symptomsText) return; const phase = getPhase(log.cycleDay); if (phases[phase]) { phases[phase].days.push(log.date); // Разбираем симптомы (разделители: запятая, точка с запятой) const symptomsList = log.symptomsText.split(/[,;]/).map(s => s.trim().toLowerCase()).filter(s => s); symptomsList.forEach(sym => { if (!phases[phase].symptoms[sym]) phases[phase].symptoms[sym] = 0; phases[phase].symptoms[sym]++; }); } }); // Сортируем фазы по порядку const phaseOrder = ['Менструация', 'Фолликулярная', 'Овуляция', 'Лютеиновая', 'Стабильная']; let html = '
'; phaseOrder.forEach(phaseName => { const data = phases[phaseName]; const totalDays = data.days.length; if (totalDays === 0) { html += `
${phaseName}: нет записей
`; return; } // Берём топ-3 симптома const sortedSymptoms = Object.entries(data.symptoms).sort((a,b) => b[1] - a[1]).slice(0, 3); const symptomText = sortedSymptoms.length ? sortedSymptoms.map(([s, count]) => `${s} (${count})`).join(', ') : 'нет симптомов'; html += `
${phaseName}
Записей: ${totalDays}
Частые симптомы: ${symptomText}
`; }); html += '

${d}

⏳ Анализ...

'; // Собираем все записи, где есть симптомы const symptomLogs = userLogs.filter(log => log.symptomsText && log.symptomsText.trim().length > 0); if (symptomLogs.length === 0) { container.innerHTML = '

Нет записей с симптомами для анализа.

'; return; } // Для каждой записи с симптомами смотрим, что было за 48 часов до неё const triggers = {}; // продукт/лекарство -> количество совпадений symptomLogs.forEach(log => { const logDate = new Date(log.date + 'T' + (log.time || '00:00')); // Ищем записи за предыдущие 48 часов (включая текущую, но без симптомов?) const twoDaysBefore = new Date(logDate); twoDaysBefore.setHours(twoDaysBefore.getHours() - 48); userLogs.forEach(possibleTrigger => { if (possibleTrigger === log) return; // пропускаем саму запись const triggerDate = new Date(possibleTrigger.date + 'T' + (possibleTrigger.time || '00:00')); if (triggerDate >= twoDaysBefore && triggerDate <= logDate) { // Если в этой записи есть продукты или лекарства const items = []; if (possibleTrigger.products) items.push(...possibleTrigger.products); if (possibleTrigger.medications) items.push(...possibleTrigger.medications); items.forEach(item => { const key = item.toLowerCase().trim(); if (!triggers[key]) triggers[key] = 0; triggers[key]++; }); } }); }); // Сортируем по частоте const sortedTriggers = Object.entries(triggers) .filter(([item, count]) => count >= 2) // минимум 2 совпадения .sort((a,b) => b[1] - a[1]) .slice(0, 10); if (sortedTriggers.length === 0) { container.innerHTML = '

'; return; } let html = '
'; sortedTriggers.forEach(([item, count]) => { html += `

`; html += '
'; container.innerHTML = html; } // ================================================================ // АВТОМАТИЧЕСКИЙ ЗАПУСК ПРИ ЗАГРУЗКЕ ВКЛАДКИ "АНАЛИТИКА" // ================================================================ // Добавляем обработчик переключения вкладок для обновления аналитики document.addEventListener('DOMContentLoaded', function() { // При переключении на вкладку analytics обновляем данные const analyticsTabBtn = document.querySelector('[data-tab="analytics"]'); if (analyticsTabBtn) { analyticsTabBtn.addEventListener('click', function() { // Даём время на рендер вкладки setTimeout(() => { renderCycleSymptomAnalysis(); renderCycleLoadCalendar(); // runAdvancedAnalytics() вызывается по кнопке }, 100); }); } // Если вкладка уже активна при загрузке (если пользователь авторизован) if (document.getElementById('analytics-tab')?.classList.contains('active')) { setTimeout(() => { renderCycleSymptomAnalysis(); renderCycleLoadCalendar(); }, 200); } });
